function calculatePayroll() { const name = document.getElementById("name").value; const basicPay = document.getElementById("basicPay").value; const allowances = document.getElementById("allowances").value; const deductions = document.getElementById("deductions").value; // Validate input if (!name || !basicPay) { alert("Please enter employee name and basic pay."); return; } // Calculate payroll const grossPay = parseInt(basicPay) + parseInt(allowances); const netPay = grossPay - parseInt(deductions); // Display pay slip const paySlip = document.getElementById("paySlip"); paySlip.innerHTML = `

Employee Name: ${name}

Basic Pay: ${basicPay}

Allowances: ${allowances || "0"}

Deductions: ${deductions || "0"}

Gross Pay: ${grossPay}

Net Pay: ${netPay}

`; }
